```
class Solution:
    def permutation(self, s: str) -> List[str]:
        c, res = list(s), []
        def dfs(x):
            if x == len(c) - 1:
                res.append(''.join(c)) #添加排列方案
                return 
            dic = set()
            for i in range(x, len(c)):
                if c[i] in dic:
                    continue # 重复，因此剪枝
                dic.add(c[i])
                c[i], c[x] = c[x], c[i]
                dfs(x+1)
                c[i], c[x] = c[x], c[i]
        dfs(0)
        return res 
```
