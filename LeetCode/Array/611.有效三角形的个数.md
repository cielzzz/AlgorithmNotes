```
class Solution:
    def triangleNumber(self, nums: List[int]) -> int:
        n = len(nums)
        nums = sorted(nums)
        cnt = 0
        for i in range(n - 1, 1, -1): #范围[2, n -1]
            j = i - 1
            k = 0
            while k < j:
                if nums[k] + nums[j] > nums[i]:
                    cnt += j - k
                    j -= 1
                else:
                    k += 1
                
        return cnt
```
